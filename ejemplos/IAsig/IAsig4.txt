#Asignaciones de tipos arrays múltiples

func none separador(){
	print(1111111111);
}

func none mostrarArray(array int[3][3] array1){
	for(int i = 0; i < 3; i+=1){
		for(int j = 0; j < 3; j+=1){
			print(array1[i][j]);
		}
	}
}

func int main(){
	
	#Arrays múltiples
	#Asignación a constante
	array int[3][3] array1 = [[0,1,2],[4,5,6],[7,8,9]];
	mostrarArray(array1);
	
	separador();
	
	#Asignación a designador
	array int[3][3] array2 = array1;
	mostrarArray(array2); 		
	return 0;
}
