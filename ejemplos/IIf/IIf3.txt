# Instrucci√≥n if-else if - else

record tRecord{
	int x; 
	real y;
}


func int main(){
	tRecord r = {10,2.3};
	int x = 1;
	if(r.x == 0){
		x = 0;
	} 
	elif(r.y == 2.3){
			if(r.y < 1){
				x = 3;
			}
			else{
				print(r.x); # Imprime 10
			}
	}
	else{
		print(14); 
	}
	
	print(x); # Imprime 1
	
	return 0;
}
